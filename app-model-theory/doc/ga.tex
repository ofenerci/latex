\chapter{Basic geometric algebra}
%
[Taken from \cite{lindner:2011}]


\section{Geometric product}
%
Let $u$, $v$ and $w$ be 3-vectors, vectors in 3-dimensional Euclidean space, $\nespace 3$. Assume a \lingo{geometric product} denoted by juxtaposition
%
\beq
  uv \not= u\iprod v \not= u\cprod v\,,
\eeq
%
which is associative
%
\beq
  u\parth{vw} = uvw = \parth{uv}w\,,
\eeq
%
but not necessarily commutative (so expect $uv \not= vu$).


\section{Normalized, antisymmetric frame}
%
Expand a vector $v$ in the (reference) \lingo{frame} $\seq{\ifvec k}{k = 1}{3} = \elset{\ifvec k}$ with components $\ivec vk$ to get
%
\beq
  v = \lsum{k = 1}{3}{\ifvec k\ivec vk}\,,
\eeq
%
where the \emph{superscript} $k$ is an \lingo{index} rather than an exponent. (Save the subscript for a future index.) Adopt \lingo{Einstein's summation convention} and shorten this to
%
\beq
  v = \ifvec k\ivec vk\,,
\eeq
%
with an \lingo{implied sum} over repeated raised and lowered indices. Latin indices always range from 1 to 3.

Assume the frame vectors $\elset{\ifvec k}$ satisfy the \lingo{abstract algebra}
%
\beq
  \ifvec k\ifvec l + \ifvec l\ifvec k = 2\iffsmet kl\,,
\eeq
%
where Kronecker's delta (in terms of Iverson's brackets)
%
\beq
  \iffsmet kl = \ivbra{k = l}
\eeq
%
is also the \lingo{flat space metric} and the $\iffsmet kl$ occupies the $k$th row and $l$th column of the square array
%
\beq
  \iffsmet kl \rightleftharpoons 
    \begin{pmatrix}
      1 & 0 & 0 \\
      0 & 1 & 0 \\
      0 & 0 & 1 \\
    \end{pmatrix}
\eeq
%
Thus, under the geometric product, the frame vectors are \lingo{antisymmetric} and \lingo{normalized}; \eg, $\ifvec 1\ifvec 2 = -\ifvec 2\ifvec 1$ and $\ifvec 3\ifvec 3 = (\ifvec 3)^2 = 1$ and so on.

Finally, shorten notation for the geometric product of frame vectors
%
\beq
  \ifvec{kl} = \ifvec k\ifvec l\,.
\eeq
